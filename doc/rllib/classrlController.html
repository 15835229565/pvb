{
  rlThread::lock();
  type = PID_T1;
  T   = _T;
  Kpp = _Kpp;
  TnP = _TnP;
  TvP = _TvP;
  Td  = _Td;
  dt = (int) (1000.0 * T);
  d0 = Kpp * ((TnP+T/2.0) / TnP) * ((TvP+T/2.0))/(Td+T/2.0);
  d1 = -2.0*Kpp * (TnP*TvP - (T/2.0)*(T/2.0)) / (TnP*(Td+T/2.0));
  d2 = Kpp*((TnP-T/2.0)/TnP)*((TvP-T/2.0)/(Td+T/2.0));
  c1 = 2*Td/(Td+T/2.0);
  c2 = -(Td-T/2.0)/(Td+T/2.0);
  rlThread::unlock();
}
